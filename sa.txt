//swapping of 2 numbers using third variable
#include<stdio.h>
int main()
{
int a=10,b=20,c;
printf("BEFORE SWAPPING A=%d B=%d",a,b);
c=a;
a=b;
b=c;
printf("AFTER SWAPPING a=%d b=%d",a,b);
}
//swapping of 2 numbers without using third variable
#include<stdio.h>
int main()
{
int a=10,b=20;
printf("before swapping A=%d B=%d",a,b);
a=a+b;
b=a-b;
a=a-b;
printf("AFTER SWAPPING a=%d b=%d",a,b);
}
